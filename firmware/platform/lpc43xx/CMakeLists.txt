	${PATH_LIBGREAT_FIRMWARE_PLATFORM_DRIVERS}/platform_config.c

	# Reset control.
	${PATH_LIBGREAT_FIRMWARE_DRIVERS}/reset.c
	${PATH_LIBGREAT_FIRMWARE_PLATFORM_DRIVERS}/platform_reset.c
